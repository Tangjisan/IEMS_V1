一、介绍

        BMP280是一款基于博世公司APSM工艺的小封装低功耗数字复合传感器，它可以测量环境温度和大气压强。气压敏感元件是一个低噪高精度高分辨率绝对大气压力压电式感应元件；温度感测元件具有低噪高分辨率特性，温度值可以对气压进行温度补偿自校正。

3.传感器数据获取流程

   BMP280的测试启动流程 :  

BMP280内部有一个温度传感器和一个压力传感器，通过 I2C接口，可读取温度和压力的数据

BMP280八位的器件地址为 111011X0，其中 X为 SDO配置状态，SDO低电平时器件地址为 11101100（0XEC），SDO高电平时器件地址为 11101110（0XEE）。从原理图可以看到，SDO默认接地，那么咱们不需要连接任何东西，在找IIC地址时就是（0xEC）

BMP280读取到的数据是芯片内部 ADC转换后的原始数值，并非最终的大气压力值。需要进行转换才能得到气压值，根据 BMP280的寄存器中的系数进行计算转换。

  166. void BMP_IIC_Send_Byte(u8 txd)
  167. {                        
  168.     u8 t;   
  169. 	BMP280_IIC_SDA_OUT(); 	    
  170.     BMP280_IIC_SCL=0;                                                                 //拉低时钟开始数据传输
  171.     for(t=0;t<8;t++)
  172.     {              
  173.         BMP280_IIC_SDA=(txd&0x80)>>7;
  174.         txd<<=1; 	  
  175. 				delay_us(2);                                                           //对TEA5767这三个延时都是必须的
  176. 				BMP280_IIC_SCL=1;
  177. 				delay_us(2); 
  178. 				BMP280_IIC_SCL=0;	
  179. 				delay_us(2);
  180.     }	 
  181. } 	    
  182. //读1个字节，ack=1时，发送ACK，ack=0，发送nACK   
  183. u8 BMP_IIC_Read_Byte(unsigned char ack)
  184. {
  185. 	unsigned char i,receive=0;
  186. 	BMP280_IIC_SDA_IN();                                                                  //SDA设置为输入
  187.     for(i=0;i<8;i++ )
  188. 		{
  189.         BMP280_IIC_SCL=0; 
  190.         delay_us(2);
  191. 				BMP280_IIC_SCL=1;
  192.         receive<<=1;
  193.         if(BMP280_READ_SDA)receive++;   
  194. 				delay_us(1); 
  195.     }					 
  196.     if (!ack)
  197.         BMP_IIC_NAck();                                                            //发送nACK
  198.     else
  199.         BMP_IIC_Ack();                                                             //发送ACK   
  200.     return receive;
  201. }

  322. // Returns temperature in DegC, resolution is 0.01 DegC. Output value of "5123" equals 51.23 DegC
  323. // t_fine carries fine temperature as global value
  324. static s32 bmp280CompensateT(s32 adcT)
  325. {
  326.     s32 var1,var2,T;
  327.  
  328.     var1=((((adcT>>3)-((s32)bmp280Cal.dig_T1<<1)))*((s32)bmp280Cal.dig_T2))>>11;
  329.     var2=(((((adcT>>4)-((s32)bmp280Cal.dig_T1))*((adcT>>4)-((s32)bmp280Cal.dig_T1)))>>12)*((s32)bmp280Cal.dig_T3))>>14;
  330.     bmp280Cal.t_fine=var1+var2;
  331. 	
  332.     T=(bmp280Cal.t_fine*5+128)>>8;
  333.  
  334.     return T;
  335. }
  336.  
  337. // 以 Q24.8 格式（24 个整数位和 8 个小数位）将压力以 Pa 为单位返回为无符号 32 位整数。
  338. // “24674867”的输出值表示 24674867/256 = 96386.2 Pa = 963.862 hPa
  339. static uint32_t bmp280CompensateP(s32 adcP)
  340. {
  341.     int64_t var1,var2,p;
  342.     var1=((int64_t)bmp280Cal.t_fine)-128000;
  343.     var2=var1*var1*(int64_t)bmp280Cal.dig_P6;
  344.     var2=var2+((var1*(int64_t)bmp280Cal.dig_P5)<<17);
  345.     var2=var2+(((int64_t)bmp280Cal.dig_P4)<<35);
  346.     var1=((var1*var1*(int64_t)bmp280Cal.dig_P3)>>8)+((var1*(int64_t)bmp280Cal.dig_P2)<<12);
  347.     var1=(((((int64_t)1)<<47)+var1))*((int64_t)bmp280Cal.dig_P1)>>33;
  348.     if (var1==0)
  349.         return 0;
  350.     p=1048576-adcP;
  351.     p=(((p<<31)-var2)*3125)/var1;
  352.     var1=(((int64_t)bmp280Cal.dig_P9)*(p>>13)*(p>>13))>>25;
  353.     var2=(((int64_t)bmp280Cal.dig_P8)*p)>>19;
  354.     p=((p+var1+var2)>>8)+(((int64_t)bmp280Cal.dig_P7)<<4);
  355.     return(uint32_t)p;
  356. }

  358. #define FILTER_NUM	5
  359. #define FILTER_A	0.1f

  361. /*限幅平均滤波法*/
  362. static void presssureFilter(float* in,float* out)
  363. {	
  364. 	static u8 i=0;
  365. 	static float filter_buf[FILTER_NUM]={0.0};
  366. 	double filter_sum=0.0;
  367. 	u8 cnt=0;	
  368. 	float deta;
  369.  
  370. 	if(filter_buf[i]==0.0f)
  371. 	{
  372. 		filter_buf[i]=*in;
  373. 		*out=*in;
  374. 		if(++i>=FILTER_NUM)	
  375. 			i=0;
  376. 	} 
  377. 	else 
  378. 	{
  379. 		if(i)
  380. 			deta=*in-filter_buf[i-1];
  381. 		else 
  382. 			deta=*in-filter_buf[FILTER_NUM-1];
  383. 		
  384. 		if(fabs(deta)<FILTER_A)
  385. 		{
  386. 			filter_buf[i]=*in;
  387. 			if(++i>=FILTER_NUM)	
  388. 				i=0;
  389. 		}
  390. 		for(cnt=0;cnt<FILTER_NUM;cnt++)
  391. 		{
  392. 			filter_sum+=filter_buf[cnt];
  393. 		}
  394. 		*out=filter_sum /FILTER_NUM;
  395. 	}
  396. }

  398. void BMP280GetData(float* pressure,float* temperature,float* asl)
  399. {
  400.     static float t;
  401.     static float p;
  402. 	
  403. 	bmp280GetPressure();
  404.  
  405. 	t=bmp280CompensateT(bmp280RawTemperature)/100.0;		
  406. 	p=bmp280CompensateP(bmp280RawPressure)/25600.0;		
  407.  
  408. 	presssureFilter(&p,pressure);
  409. 	*temperature=(float)t;                                                   /*单位度*/
  410. 	*pressure=(float)p ;	                                                   /*单位hPa*/	
  411. 	
  412. 	*asl=bmp280PressureToAltitude(pressure);	                               /*转换成海拔*/	
  413. }
  414.  
  415. #define CONST_PF 0.1902630958	                                               //(1/5.25588f) Pressure factor
  416. #define FIX_TEMP 25				                                               // Fixed Temperature. ASL is a function of pressure and temperature, but as the temperature changes so much (blow a little towards the flie and watch it drop 5 degrees) it corrupts the ASL estimates.
  417. 								                                               // TLDR: Adjusting for temp changes does more harm than good.
  418. /*
  419.  * Converts pressure to altitude above sea level (ASL) in meters
  420. */
  421. static float bmp280PressureToAltitude(float* pressure/*, float* groundPressure, float* groundTemp*/)
  422. {
  423.     if (*pressure>0)
  424.     {
  425.         return((pow((1015.7f/ *pressure),CONST_PF)-1.0f)*(FIX_TEMP+273.15f))/0.0065f;
  426.     }
  427.     else
  428.     {
  429.         return 0;
  430.     }
  431. }
