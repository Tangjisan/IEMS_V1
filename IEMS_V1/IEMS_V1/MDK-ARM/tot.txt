#include "sensor_PMS5003.h"

namespace n_PMS5003{
    auto _set_pin = _PMS5003_SET_Pin;
    auto _set_port = _PMS5003_SET_GPIO_Port;
    auto _reset_pin = _PMS5003_RESET_Pin;
    auto _reset_port = _PMS5003_RESET_GPIO_Port;
    auto _husart = huart1;

    uint8_t* _buffer_rx = nullptr;
    uint8_t _buffer_rx_sz = 32;    // 硬件需32字节接收
    uint8_t _buffer_rx_idx = 0;    // 用于记录当前处理到那个数据

    const uint8_t _START0_CMD = 0x42;
    const uint8_t _START1_CMD = 0x4d;
    const uint16_t _FRAMELENTH_CMD = 0x1C;

    data_t _data = {0};
    uint8_t _data_flag = 0;     // 0:数据异常，不能读取，1:数据正常
    uint8_t _is_use_driver = 0; // 0:未用本硬件，1:使用本硬件

    void sleep(void){
        HAL_GPIO_WritePin(_set_port, _set_pin, GPIO_PIN_RESET);
    }
    void wake(void){
        HAL_GPIO_WritePin(_set_port, _set_pin, GPIO_PIN_SET);
    }
    void reset(void){
        HAL_GPIO_WritePin(_reset_port, _reset_pin, GPIO_PIN_RESET);
        HAL_Delay(10);
        HAL_GPIO_WritePin(_reset_port, _reset_pin, GPIO_PIN_SET);
    }

    HAL_StatusTypeDef init(void){
        sleep();
        reset();
        wake();

        _buffer_rx = new(std::nothrow) uint8_t[_buffer_rx_sz]();
        if(_buffer_rx == nullptr){
            deinit();
            return HAL_ERROR;
        }
        return HAL_OK;
    }
    HAL_StatusTypeDef deinit(void){
        sleep();
        delete[] _buffer_rx;
        _buffer_rx = nullptr;
        return HAL_OK;
    }

    bool parseData(uint8_t* buf, data_t* pms_data){
        // 计算帧长度
        uint16_t frameLength = ((buf[2] << 8) | buf[3]);
        if(frameLength != _FRAMELENTH_CMD)  return false;
        
        // 计算校验码
        uint16_t checksum = 0;
        for(int i = 0; i < 30; ++i){
            checksum += buf[i];
        }
        uint16_t frame_check = (buf[30] << 8) | buf[31];
        if(checksum != frame_check) return false;
        
        pms_data->pm1_0 = (buf[4] << 8) | buf[5];
        pms_data->pm2_5 = (buf[6] << 8) | buf[7];
        pms_data->pm10  = (buf[8] << 8) | buf[9];

        return true;
    }

    namespace n_cb{

        void receive(void){
            if(n_get::is_use_driver() == false) return;
            uint8_t data = 0;
            HAL_UART_Receive(&_husart, &data, 1, 10);
            // 检查起始符
            if(_buffer_rx_idx == 0 && data != _START0_CMD)  return;
            if(_buffer_rx_idx == 1 && data != _START1_CMD){
                _buffer_rx_idx = 0;
                return;
            }

            _buffer_rx[_buffer_rx_idx++] = data;

            if(_buffer_rx_idx >= _buffer_rx_sz){
                _buffer_rx_idx = 0;
                if(parseData(_buffer_rx, &_data) == false)  _data_flag = 0;
                else _data_flag = 1;
            }
        }
    } // namespace n_cb
    namespace n_get{
        bool is_data_ready(void){
            return _data_flag == 1;
        }
        bool is_use_driver(void){
            return _is_use_driver == 1;
        }
        uint16_t pm1_0(void){
            return _data.pm1_0;
        }
        uint16_t pm2_5(void){
            return _data.pm2_5;
        }
        uint16_t pm10(void){
            return _data.pm10;
        }
    } // namespace n_get
    
}

HAL_StatusTypeDef init(void){
    // 1. 硬件复位与唤醒，增加初始化延迟
    sleep();         // 先休眠（确保初始状态一致）
    reset();         // 复位传感器
    HAL_Delay(200);  // 关键：等待复位完成（至少100ms）
    wake();          // 唤醒（SET引脚高电平）
    HAL_Delay(1000); // 等待传感器硬件就绪（供电稳定）

    // 2. 发送模式切换指令，增加指令间隔
    n_cmd::normal_mode();       // 先切换到正常模式（退出待机）
    HAL_Delay(100);             // 等待指令响应
    auto ret = n_cmd::transform_auto_mode(); // 切换到主动模式
    HAL_Delay(100);             // 等待模式切换生效

    // 3. 30秒预热（传感器稳定输出的必要条件）
    HAL_Delay(30000);

    // 4. 初始化缓冲区并启动接收
    _buffer_rx = new(std::nothrow) uint8_t[_buffer_rx_sz]();
    if(_buffer_rx == nullptr){
        deinit();
        return HAL_ERROR;
    }

    _is_use_driver = 1;
    _buffer_rx_idx = 0;
    _data_flag = 0;

    // 启动中断接收
    HAL_UART_Receive_IT(&_husart, &_buffer_rx[_buffer_rx_idx], 1);
    
    // 指令发送成功的LED指示（确认硬件连接正确）
    if(ret == HAL_OK)   HAL_GPIO_WritePin(LED_P5_GPIO_Port, LED_P5_Pin, GPIO_PIN_RESET);
    
    return HAL_OK;
}
